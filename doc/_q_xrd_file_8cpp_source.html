230 bool QXrdFilePrivate::exists() const {
231  XrdClientAdmin admin(qPrintable(this->_fileName.left(this->_fileName.lastIndexOf("/"))));
232  if (!admin.Connect())
233  return false;
234  vecBool vb;
235  vecString vs;
236  XrdOucString name(qPrintable(this->_fileName.split("//").last().prepend("/")));

238  admin.ExistFiles(vs, vb);
239  return vb.At(0);
